\begin{frame}{Thông điệp chính}
  \begin{block}{Luận điểm}
  Một DSS nhỏ nhưng đầy đủ có thể chuyển dữ liệu đơn hàng pizza thành hàng đợi
  ưu tiên và gợi ý hành động vận hành.
  \end{block}
  \begin{itemize}
    \item Dữ liệu Kaggle thật: 1.004 đơn hàng, 25 cột gốc, 37 cột sau FE.
    \item Kiểm soát leakage: không dùng duration/delay sau giao hàng làm feature.
    \item Audit dữ liệu synthetic: 12 cảnh báo data-realism.
    \item Thêm reverse-engineering generator, behavior, forecasting, recommendation, staffing và transportation scenario.
  \end{itemize}
\end{frame}

\begin{frame}{Tham khảo bài toán vận tải}
  \begin{block}{Ý tưởng}
  Lấy các đơn có rủi ro cao trong queue, tạo fleet tài xế giả định và giải
  bài toán assignment tối thiểu chi phí.
  \end{block}
  \begin{itemize}
    \item Đơn hàng là dữ liệu thật từ queue.
    \item Driver/capacity là kịch bản mô phỏng vì dataset gốc không có.
    \item Objective: giảm chi phí gán dựa trên distance, traffic, risk và base location.
    \item Output: \texttt{reports/metrics/transport\_assignment.csv}.
  \end{itemize}
\end{frame}

\begin{frame}{Giới hạn và hướng phát triển}
  \begin{itemize}
    \item Dataset nhỏ và có tính mô phỏng.
    \item Có 12 cảnh báo data-realism; không overclaim kết quả kinh doanh.
    \item Brand/preference trend có artifact từ generator, không phải kết luận thị trường thật.
    \item Chronological holdout không có delayed sample, nên bản học thuật dùng stratified split.
    \item Forecast MAPE 45,7\%, chỉ là demo planning.
    \item Driver/fleet là mô phỏng, chưa phải dữ liệu điều phối thật.
    \item Priority chưa được tối ưu bằng cost/SLA thật.
    \item Hướng phát triển: thu thập driver logs, SLA, route distance, cost và kiểm thử theo thời gian.
  \end{itemize}
\end{frame}
