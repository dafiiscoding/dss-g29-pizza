\begin{frame}{8. Độ đo đánh giá: định nghĩa \& vì sao chọn}
    \begin{columns}[T]
        \column{0.56\linewidth}
        \blk{Các thước đo dùng trong báo cáo}
        \begin{itemize}\footnotesize
            \item \textbf{Accuracy} — tỷ lệ đoán đúng trên tổng đơn. \emph{Bẫy}: đoán mọi đơn ``đúng giờ'' vẫn được $79\%$.
            \item \textbf{Balanced Accuracy} — trung bình độ nhạy hai lớp $=\tfrac{1}{2}(\text{Recall}_{\text{trễ}}+\text{Recall}_{\text{đúng giờ}})$; không bị lớp đa số chi phối.
            \item \textbf{Precision} $=\tfrac{TP}{TP+FP}$ — trong các đơn \emph{bị báo trễ}, bao nhiêu trễ thật.
            \item \textbf{Recall} (độ nhạy) $=\tfrac{TP}{TP+FN}$ — bắt được bao nhiêu đơn trễ thật.
            \item \textbf{F2} — trung bình điều hoà P \& R nhưng \textbf{coi trọng Recall gấp $\sim$2 lần}.
            \item \textbf{MCC} — tương quan giữa nhãn thật và nhãn dự đoán; $1=$ hoàn hảo, $0=$ đoán mò, $-1=$ ngược hoàn toàn; dùng \textbf{cả 4 ô} ma trận nhầm lẫn.
        \end{itemize}

        \column{0.44\linewidth}
        \blk{Vì sao chọn F2 \& MCC?}
        \begin{itemize}\footnotesize
            \item Dữ liệu lệch lớp ($\sim21\%$ trễ) $\Rightarrow$ Accuracy gây ảo giác.
            \item Bỏ sót đơn trễ (FN) tốn kém hơn báo nhầm (FP) $\Rightarrow$ \textbf{ưu tiên Recall} $\Rightarrow$ chọn \textbf{F2} làm chỉ số tối ưu chính.
            \item \textbf{MCC} làm ``trọng tài'' tổng thể: nếu mô hình chỉ giỏi đoán lớp đa số nhưng sai lớp trễ, MCC sẽ không cao.
        \end{itemize}
        \vspace{2pt}
        {\scriptsize
        \[ F_2=\frac{5\,P\,R}{4P+R},\;\;
           \text{MCC}=\frac{TP\cdot TN-FP\cdot FN}{\sqrt{(TP{+}FP)(TP{+}FN)(TN{+}FP)(TN{+}FN)}} \]
        }
    \end{columns}
\end{frame}

\begin{frame}{10B. Giao thông: ý nghĩa mô hình \& điều phối}
    \begin{columns}[T]
        \column{0.5\linewidth}
        \blk{Vì sao đưa traffic vào feature?}
        \begin{itemize}
            \item Traffic biết được \textbf{trước khi giao}, nên hợp lệ cho dự báo sớm.
            \item Mức traffic phản ánh thời gian di chuyển, khả năng kẹt xe và độ bất định ETA.
            \item Được đưa vào mô hình bằng \textbf{one-hot encoding}; không ép giả định khoảng cách tuyến tính cứng giữa Low/Medium/High.
        \end{itemize}

        \column{0.48\linewidth}
        \blk{Vì sao traffic có trong DSS Risk Score?}
        \begin{itemize}
            \item Ngoài xác suất mô hình, DSS cộng thêm \textbf{traffic pressure} để người điều phối thấy nguyên nhân rủi ro.
            \item Chính sách điểm: Low=20, Medium=60, High=100; trọng số 0,15 trong Risk Score.

\begin{frame}{15. Feature Engineering: làm gì \& vì sao}

        \blk{Vì sao làm vậy?}
        \begin{itemize}\footnotesize
            \item Cỡ bánh có thứ tự tự nhiên $\Rightarrow$ mã thứ bậc giữ quan hệ đơn điệu, ít cột hơn one-hot.

            \item Bỏ biến trùng $\Rightarrow$ tránh đa cộng tuyến: \textbf{17 biến $\to$ 12 biến} gọn và dễ giải thích hơn.
        \end{itemize}

\begin{frame}{24. Điểm rủi ro: công thức tính (0--100)}
    \begin{columns}[T]
        \column{0.5\linewidth}
        \blk{Cách nhóm đặt trọng số}
        \begin{itemize}
            \item \textbf{55\% cho mô hình:} xác suất trễ vẫn là tín hiệu chính.
            \item \textbf{45\% còn lại:} cộng thêm traffic, cự ly, giờ cao điểm, độ phức tạp và cuối tuần.
            \item Các trọng số cộng bằng 1 và được đặt thủ công để dễ đọc, không phải hệ số học từ mô hình.
        \end{itemize}

        \column{0.48\linewidth}
        \blk{Công thức tính toán (0--100)}
        {\scriptsize
        \begin{equation*}
        \begin{aligned}
            \text{Rủi ro} = {} & 0.55 \times S_{\text{mô hình}} \\
            & + 0.15 \times S_{\text{giao thông}} \\
            & + 0.12 \times S_{\text{quãng đường}} \\
            & + 0.08 \times S_{\text{cao điểm}} \\
            & + 0.06 \times S_{\text{phức tạp}} \\
            & + 0.04 \times S_{\text{cuối tuần}}
        \end{aligned}
        \end{equation*}
        }
        \vspace{-0.15cm}
        \begin{itemize}\scriptsize
            \item Dự đoán mô hình chiếm \textbf{55\%} (bằng chứng chính); \textbf{45\%} còn lại là các chỉ số vận hành thực tế.
        \end{itemize}
    \end{columns}
